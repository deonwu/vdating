/**
 * 用户排名的数据列表。
 * 
 * @method ranking_data_list
 * @param {String} cate -- 排名方式，可选值从 ranking_cate_list 取得。
 * 
 * @return {Object} 用户列表信息
 */
AV.Cloud.define("ranking_data_list", function(request, response) {
	  var Ranking = AV.Object.extend("RankingList");
	  
	  var result = {'status': 'ok'};
	  var data = {};
	  
	  var cate = request.params.cate || 'g_year';
	  var cate = cate.split("_");
	  
	  var cur = new Date();
	  var month = new Date(cur.getTime() - 1000 * 60 * 60 * 24 * 3);
	  var year = new Date(cur.getTime() - 1000 * 60 * 60 * 24 * 2 * 30);
	  
	  var c_year = month.getFullYear() + "";
	  var c_month = month.getMonth() + 1;
	  c_month = c_month < 10 ? "0" + c_month : c_month + ""; 

	  var r_month = c_year + c_month;
	  var r_year = year.getFullYear() + "";
	  
	  var r_date = cate[1] == 'month' ? r_month : r_year;
	  
	  var r_field = cate[0] + "_" + r_date;
	  
	  console.log("ranking field:" + r_field);
	  
	  var r_unit = cate[0] == 'i' ? '邀请积分' : '鲜花';
	  
	  new AV.Query(Ranking).include("user")
	  .greaterThan(r_field, 1)
	  .limit(100)
	  .descending(r_field)
	  .find()
	  .then(function(results){
		  
		  var users = [];
		  for (var i in results) {
			  var item = results[i];
			  
			  var user = {};
			  u = item.get('user');
			  
			  user.ranking_data = item.get(r_field);
			  user.user_id = u.id;
			  user.head_img = u.get('headImg')['_url']; //get('url');
			  user.user_nick = u.get('name'); //.url;
			  user.ranking_unit = r_unit;
			  
			  users.push(user);
		  }
		  
		  result['status'] = 'ok';
		  result['data'] = users;		  
	  })
	  .then(function(obj){
		  response.success(result);	  
	  }, function(obj){
		  console.log("error:" + (obj ? obj.toString(): ""));
		  
		  if(result.status == 'ok'){
			  result.status = 'err';
		  }
		  result.data = obj;
		  response.success(result);
	  })
	  ;	  
});

/** 
 * 修改约会会话的状态，例如: 接受，拒绝等。
 * 
 * @method dating_session_update_status
 * @param {String} session_id 会话ID
 * @param {String} action 操作从dating_session_status接口返回的操作列表。(reject|accept|confirm|complain|success|falied)
 *	
 * @return {Object} 放回状态信息。
 */

AV.Cloud.define("dating_session_update_status", function(request, response) {
	
	  var session = AV.Object.extend("DatingSession");
	  
	  var log = AV.Object.extend("DatingActionLog");
	  
	  var result = {'status': 'ok'};
	  var data = {};
	  
	  var st = {"accept": "已接受", "reject": "已拒绝", "success": "约会成功",
			  "failed": "约会失败", "cancel": "约会取消"};
	  
	  var actions = {
			  'new_owner_reject': {'s': 'reject'},
			  'new_owner_accept': {'s': 'accept'},

			  'new_request_cancel': {'s': 'cancel'},

			  'accept_owner_success': {'s': 'success_owner'},	//发起者标志成功。
			  'accept_owner_failed': {'s': 'failed_owner'},   	//发起者标志失败。

			  'success_owner_request_confirm': {'s': 'confirm_ok'},	//确认
			  'success_owner_request_complain': {'s': 'complain_ok_request'},	//投诉

			  'failed_owner_request_confirm': {'s': 'confirm_failed_owner'},   	//确认取消
			  'failed_owner_request_complain': {'s': 'complain_failed_owner'},   //投诉

			  'accept_request_success': {'s': 'success_request'},	//发起者标志成功。
			  'accept_request_failed': {'s': 'failed_request'},   	//发起者标志失败。

			  'success_request_owner_confirm': {'s': 'confirm_ok'},	//确认
			  'success_request_owner_complain': {'s': 'complain_ok_owner'},	//投诉

			  'failed_request_owner_confirm': {'s': 'confirm_failed_request'},   	//确认取消
			  'failed_request_owner_complain': {'s': 'complain_failed_request'},   //投诉
	  };

	  var status_label = {"accept": "请求已接受，等待见面", 
  			  "reject": "请求已拒绝", 
  			  "success": "见面约会成功",
  			  "failed": "约会爽约没有成功", 
  			  "success_owner": "发起者确认见面", 
  			  "failed_owner": "参与者爽约未见面", 

  			  "success_request": "参与者确认见面", 
  			  "failed_request": "发起者爽约未见面", 

  			  "confirm_failed_owner": "已确认参与者爽约未见面", 
  			  "confirm_failed_request": "已确认发起者爽约未见面", 
  			  
  			  "complain_ok_request": "投诉处理中", 
  			  "complain_ok_owner": "投诉处理中",
  			  "complain_failed_request": "投诉处理中", 
  			  "complain_failed_owner": "投诉处理中", 
  			  
  			  "confirm_ok": "双方确认见面成功",
  			  
  			  "cancel": "约会已取消",
  			  "new": "等待应答"
				  };	  
	  
	  var curUser = AV.User.current();
	  
	  if(!curUser){
		  var curUser = new AV.User();
		  curUser.id = request.params.user_id;
	  }
	  
	  function process_status_change(ctx, action){
		  var session = ctx.session;
		  
		  var cur_st = session.get('status');
		  var key = cur_st + "_" + ctx.user_type + "_" + action;
		  
		  var new_st = actions[key];
		  
		  new_st = new_st ? new_st['s'] : cur_st;
		  
		  console.log("action key:" + key + ", new st:" + new_st);
		  
		  data.old_status = cur_st;
		  if(cur_st != new_st){
			  data.new_status = new_st;	
			  data.message = "状态修改为," + status_label[new_st];
			  session.set("status", new_st);
			  return session.save();
		  }else {
			  return AV.Promise.error();
		  }
	  }
	  
	  
	  var session_id = request.params.session_id || "no_sid";
	  //curUser = curUser || 
	  new AV.Query(session).include("topic.user_obj").get(session_id)
	  .then(function(obj){
		  if(obj){
			  data.session = obj;
			  var t = obj.get("topic");
			  var owner = t ? t.get("user_obj") : null;
			  var send = obj.get("send_user");
			  
			  console.log("owner:" + owner + ", send:" + send + ", cur:" + curUser);
			  if(owner && send){
				  console.log("onwer:" + owner.id + ", send:" + send.id + ", cur:" + curUser.id);
				  
				  data.onwer_id = owner.id;
				  data.request_id = send.id;
				  data.user_type = owner.id == curUser.id ? "owner" : "request";
				  if(owner.id == curUser.id || send.id == curUser.id){
					  return process_status_change(data, request.params.action)
					  	.then(function(e){
							return new log().save({"session": obj, 
					  			  "user": curUser, 
					  			  "action": request.params.action,
					  			  "new_status": data.new_status,
					  			  "old_status": data.old_status
					  			 });					  		
					  	}, function(e){
							  result.status = 'err';
							  result.code = 'no_change';	
							  result.cur_status = data.old_status;	
					  	});
				  }else {
					  result.status = 'err';
					  result.code = 'not_right';
					  result.cur_user = curUser.id;				  
				  }
			  }
		  }else {
			  result.status = 'err';
			  result.code = 'not_found_from_session';
		  }
	  })
	  .then(function(obj){
		  if(result.status == 'ok'){
			  var to_user = data.onwer_id == curUser.id ? data.request_id : data.onwer_id
			  var ss = AV.Cloud.run("send_message", {'to_user': to_user,
				  'from_user': curUser.id, 
				  'msg_type': 'dating_change',
				  'message_param': data.new_status,
				  'message': data.message,
				  'bi_dir': true
				  });
			  result.status = 'ok';
			  result.new_status = data.session.get('status');
			  
			  return AV.Promise.as();
		  }else {
			  return AV.Promise.error();			  
		  }
	  })
	  .then(function(obj){
		  response.success(result);	  
	  }, function(obj){
		  console.log("error:" + (obj ? obj.toString(): ""));
		  
		  if(result.status == 'ok'){
			  result.status = 'err';
		  }
		  result.data = obj;
		  response.success(result);
	  })
	  ;	  
});

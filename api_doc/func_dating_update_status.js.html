/** 
 * 修改约会活动的状态。
 * 
 * @method dating_topic_update_status
 * @param {String} topic_id 约会主题ID
 * @param {String} action 操作(remove)
 *	
 * @return {Object} 放回状态信息。
 */

AV.Cloud.define("dating_topic_update_status", function(request, response) {
	  var topic = AV.Object.extend("DatingTopic");
	  
	  var result = {'status': 'ok'};
	  var data = {};
	  
	  var curUser = AV.User.current();
	  
	  if(!curUser){
		  var curUser = new AV.User();
		  curUser.id = request.params.user_id;
	  }
	  	  
	  var topic_id = request.params.topic_id || "no_tid";
	  var action = request.params.action || "remove";
	  
	  var st = {'remove': 'removed'};
	  
	  var new_st = st[action] || 'active';
	  
	  new AV.Query(topic).get(topic_id)
	  .then(function(obj){
		  if(obj){
			  console.log("cur topic status:" + obj.get('status') + ", tid:" + topic_id);
			  if(obj.get('status') != new_st){
				  result.new_status = new_st;
				  return obj.save({'status': new_st});
			  }else {
				  result.code = "no_change";				  
			  }
		  }else {
			  result.status = 'err';
			  result.code = 'not_found_dating_topic';
		  }
	  })
	  .then(function(obj){
		  response.success(result);	  
	  }, function(obj){
		  console.log("error:" + (obj ? obj.toString(): ""));
		  
		  if(result.status == 'ok'){
			  result.status = 'err';
		  }
		  result.data = obj;
		  response.success(result);
	  })
	  ;	  
});

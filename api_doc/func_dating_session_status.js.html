/** 
 * 查询一个约会当前的状态。和可以执行的操作。
 * 
 * @method dating_session_status
 * @param {String} session_id 会话ID
 *	
 * @return {Object} 放回状态信息。
 */
  
AV.Cloud.define("dating_session_status", 
		function(request, response) {
	  var session = AV.Object.extend("DatingSession");
	  
	  var result = {'status': 'ok'};
	  var actions = {'new_owner': [{'action': 'accept', 'label':'接受'},
	                               {'action': 'reject', 'label':'拒绝'},
	                               ],
	                 'new_request': [{'action': 'none', },
	          	                   	 {'action': 'cancel', 'label':'取消'},
	          	                   	],
	          		 'accept_owner': [{'action': 'success', 'label':'已见面'},
	          	                 	  {'action': 'failed', 'label':'TA爽约'},
	         	                   	 ],	                 
	         		 'accept_request': [{'action': 'success', 'label':'已见面'},
	         		                  {'action': 'failed', 'label':'TA爽约'},
	         		         	     ],
	         		 'success_owner_request': [{'action': 'confirm', 'label':'确认'},
	         		                           {'action': 'complain', 'label':'投诉'},
			         		         ],
	         		 'failed_owner_request': [{'action': 'confirm', 'label':'确认'},
	         		                           {'action': 'complain', 'label':'投诉'},
			         		         ],
	         		 'success_request_owner': [{'action': 'confirm', 'label':'确认'},
	         		                           {'action': 'complain', 'label':'投诉'},
			         		         ],
	         		 'failed_request_owner': [{'action': 'confirm', 'label':'确认'},
	         		                           {'action': 'complain', 'label':'投诉'},
			         		         ],			         		         
	  				};
	  
	  var data = {};
	  
	  var curUser = AV.User.current();
	  
	  if(!curUser){
		  var curUser = new AV.User();
		  curUser.id = request.params.user_id;
	  }
	  
	  var session_id = request.params.session_id || "no_sid";
	  new AV.Query(session).include("topic.user_obj").get(session_id)
	  .then(function(obj){
		  if(obj){
			  data.session = obj;
			  var t = obj.get("topic");
			  
			  var owner = t ? t.get("user_obj") : null;
			  
			  var send = obj.get("send_user");
			  
			  console.log("owner:" + owner + ", send:" + send + ", cur:" + curUser);			  
			  
			  if(owner){
				  console.log("onwer:" + owner.id + ", send:" + send.id + ", cur:" + curUser.id);
				  data.onwer_id = owner.id;
				  data.request_id = send.id;
				  if(owner.id == curUser.id || send.id == curUser.id){
					  var st = obj.get("status");
					  var relation = owner.id == curUser.id ? 'owner' : 'request';
					  result.actions = actions[st + "_" + relation];
					  result.dating_status = st;
					  result.status_label = status_label[st];				  
				  }else {
					  result.status = 'err';
					  result.code = 'not_right';
					  result.cur_user = curUser.id;	
					  result.cur_username = curUser.get('username');
					  result.session_id = session_id;				  
				  }
			  }else {
				  result.status = 'err';
				  result.code = 'topic_error';
				  result.status_label = "约会已经失效或删除。";	
			  }
		  }else {
			  result.status = 'err';
			  result.code = 'not_found_session';
		  }
	  })
	  .then(function(obj){
		  response.success(result);	
	  }, function(obj){
		  console.log("error:" + (obj ? obj.toString(): ""));
		  
		  if(result.status == 'ok'){
			  result.status = 'err';
		  }
		  result.data = obj;
		  response.success(result);
	  })
	  ;
});

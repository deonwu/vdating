/**
 * 保存用户的邀请关系，已经邀请过了，就忽略。没有邀请过，就保存邀请关系。
 * 
 * @method invite_friend
 * @param {String} tel_no 已经邀请了的用户手机号码。
 * 
 * @return {Object} 对象列表
 */

AV.Cloud.define("invite_friend", function(request, response) {
	  var Invite = AV.Object.extend("InviteFriend");
	  
	  var result = {'status': 'ok'};
	  var data = {};
	  
	  var curUser = AV.User.current();
	  
	  if(!curUser){
		  var curUser = new AV.User();
		  curUser.id = request.params.user_id;
	  }
	  
	  //curUser = curUser || 
	  new AV.Query(Invite).include("friend_list").equalTo("owner", curUser)
	  .first()
	  .then(function(obj){
		  if(obj){
			  var t = obj.get("friend_list");
			  
			  if(t && t.indexOf(request.params.tel_no) >= 0){
				  result.status = 'ok';
				  result.code = 'invited';
			  }else {
				  obj.add('friend_list', request.params.tel_no);
			  }

			  var t = obj.get("blacklist");
			  
			  if(t && t.indexOf(request.params.tel_no) >= 0){
			  }else {
				  obj.add('blacklist', request.params.tel_no);
			  }	  
			  
			  return obj.save();
		  }else {
			  return new Invite().save({'owner': curUser, 
				  					  'friend_list':[request.params.tel_no, ]});
		  }
	  })
	  .then(function(obj){
		  response.success(result);	  
	  }, function(obj){
		  console.log("error:" + (obj ? obj.toString(): ""));
		  
		  if(result.status == 'ok'){
			  result.status = 'err';
		  }
		  result.data = obj;
		  response.success(result);
	  })
	  ;	  
});
